# Install required packages (run once)
required_packages <- c("dplyr", "ggplot2", "readr")
missing_packages <- required_packages[!(required_packages %in% installed.packages()[, "Package"])]
if (length(missing_packages) > 0) install.packages(missing_packages)
data(mtcars)

str(mtcars)

# Fit linear regression model
lin_model <- lm(mpg ~ wt, data = mtcars)

# View summary
summary(lin_model)

mtcars$predicted <- predict(lin_model)

head(mtcars[, c("wt", "mpg", "predicted")])

plot(mtcars$wt, mtcars$mpg,
     xlab = "Weight (1000 lbs)",
     ylab = "Miles Per Gallon",
     main = "Linear Regression",
     pch = 16)

abline(lin_model, col = "blue", lwd = 2)

# install.packages("ggplot2") # if needed
library(ggplot2)

ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_point(color = "darkgreen") +
  geom_smooth(method = "lm", se = TRUE, color = "blue") +
     labs(title = "Linear Regression: MPG vs Weight",
                x = "Weight (1000 lbs)",
                y = "Miles Per Gallon")
